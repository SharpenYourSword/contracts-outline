\section{Structure of a Contract}

\begin{enumerate}
    \item Parties.
    \begin{enumerate}
        \item Negotiation/barganing.
        \begin{enumerate}
            \item Fraud/duress.
        \end{enumerate}
        \item Offer/acceptance.
    \end{enumerate}
    \item Contract.
    \begin{enumerate}
        \item Consideration.
        \item Reliance.
        \item Unconscionability.
        \item Types:
        \begin{enumerate}
            \item Unilateral.
            \item Bilateral.
            \item Multilateral.
            \item Adhesion.
            \item Quasi.
        \end{enumerate}
    \end{enumerate}
    \item Performance.
    \begin{enumerate}
        \item Perfect.
        \item Flawed.
    \end{enumerate}
    \item (Modifications.)
    \item Damages.
\end{enumerate}
